AngularJS 模块

    var app = angular.module('myApp', []);

AngularJS 控制器

    var app = angular.module('myApp', []);
    app.controller('myCtrl', function($scope) {
        $scope.names = ['张三', '李四', '王五'];
        $scope.peoples = [
            {name: '张三', country: '中国'},
            {name: '李四', country: '美国'},
            {name: '王五', country: '英国'},
        ];
    })
