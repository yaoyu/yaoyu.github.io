    var res = 1;

    var length2 = length * 0.5;
    length--;
    for(var i=0; i<length2; ++i){
        if (str[i] != str[length-i])

function findLongestWord(str) {
    var parts = str.split(' ');
    var max = 0;
    var len = 0;
    for (var x in parts){
        len = parts[x].length;
        if (max < len){
            max = len;
        }
    }
    return max;
}

设置首字母大写算法

function UpperFirst(word){
    return word[0].toUpperCase() + word.slice(1).toLowerCase();
}

function titleCase(str) {
    var parts = str.split(' ');
    var newWords = [];
    for (var x in parts){
        newWords.push(UpperFirst(parts[x]));
    }
    return newWords.join(' ');
}

titleCase("I'm a little tea pot");

function largest(arr){
    var max = -99999999;
    for (var x in arr){
        if (max < arr[x]){
            max = arr[x];
        }
    }
    return max
}
function largestOfFour(arr) {
    var newArray = [];

数组分割算法 slice

function chunk(arr, size) {
    var res = [];
    while(arr.length > size){
        res.push(arr.slice(0, size));
        arr = arr.slice(size);
    }
    if (arr.length > 0){
        res.push(arr);
    }
    return res;
}

chunk(["a", "b", "c", "d"], 2);

数组排序并插入值算法 sort(function)

function sortNumber(a,b)
{
    return a - b;
}

function where(arr, num) {
    var index = -1;
    arr.sort(sortNumber);
    for (var x = 0; x < arr.length; ++x){
        if (arr[x] > num){
            index = x;
            arr.splice(index, 0, num);
            break;
        } else if (arr[x] == num) {
            index = x;
            break;
        }
    }
    if (index == -1){
        index = arr.length;
        arr.push(num);
    }
    return index;
}

where([40, 60], 50);

位移密码算法

String.prototype.charCodeAt()
String.fromCharCode()

一个常见的案例就是ROT13密码，字母会移位13个位置。由'A' ↔ 'N', 'B' ↔'O'，以此类推。

写一个ROT13函数，实现输入加密字符串，输出解密字符串。

所有的字母都是大写，不要转化任何非字母形式的字符(例如：空格，标点符号)，遇到这些特殊字符，就跳过它们。

function rot13(str) { // LBH QVQ VG!
    var res = []
    var datas = str.split('');
    for (var i = 0; i < datas.length; ++i){
        var item = datas[i];
        var code = item.charCodeAt();
        if ((item >= 'a' && item <= 'z') || (item >= 'A' && item <= 'Z')) {
            if (code > 77)
                res.push(String.fromCharCode(code - 13));
            else
                res.push(String.fromCharCode(code + 13));
        } else {
            res.push(item);
        }
    }
    return res.join('');
}
