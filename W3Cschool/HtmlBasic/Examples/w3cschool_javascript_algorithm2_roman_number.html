JavaScript算法 数字转罗马数字

罗马数字是阿拉伯数字传入之前使用的一种数码。罗马数字采用七个罗马字母作数字、即Ⅰ（1）、X（10）、C（100）、M（1000）、V（5）、L（50）、D（500）。记数的方法：
  1. 相同的数字连写，所表示的数等于这些数字相加得到的数，如 Ⅲ=3；
  2. 小的数字在大的数字的右边，所表示的数等于这些数字相加得到的数，如 Ⅷ=8、Ⅻ=12；
  3. 小的数字（限于 Ⅰ、X 和 C）在大的数字的左边，所表示的数等于大数减小数得到的数，如 Ⅳ=4、Ⅸ=9；
  4. 在一个数的上面画一条横线，表示这个数增值 1,000

以下是一段示例代码:

	var keyNum = {1: 'I',
        5: 'V',
        10: 'X',
        50: 'L',
        100: 'C',
        500: 'D',
        1000: 'M'
    };

	function repeat(str, num){
	    var res = []
	    for (var i=0; i < num; ++i){
	        res.push(str);
	    }
	    return res.join('');
	}

	function n1(num){
	    return repeat('I', num);
	}

	function units(num){
		if (num === 0){
			return '';
		} else if (num < 4)
			return n1(num);
		else if  (num == 4)
			return n1(1) + keyNum[num+1];
		else if (num < 9)
			return keyNum[5] + n1(num - 5);
		else if (num < 10)
			return n1(1) + keyNum[num+1];
	}

	function n10(num = 1){
	    return repeat('X', num);
	}

	function ten10(num){
		if (num === 0){
			return '';
		} else if (num < 4){
	        return n10(num);
	    } else if (num == 4) {
	        return n10(1) + keyNum[50];
	    } else if (num < 9){
	        return keyNum[50] + n10(num - 5)
	    } else if (num == 9){
	        return n10(1) + keyNum[100];
	    }
	}

	function n100(num = 1){
	    return repeat('C', num);
	}

	function hundreds(num){
		if (num === 0){
			return '';
		} else if (num < 4){
	        return n100(num);
	    } else if (num == 4) {
	        return n100(1) + keyNum[500];
	    } else if (num < 9){
	        return keyNum[500] + n100(num - 5);
	    } else if (num == 9){
	        return n100(1) + keyNum[1000];
	    }
	}

	function n1000(num = 1){
	    return repeat('M', num);
	}

	function tenhundreds(num){
		if (num === 0){
			return '';
		} else if (num < 4){
	        return n1000(num);
	    } else if (num == 4) {
	        return n1000(1) + keyNum[5000];
	    } else if (num < 9){
	        return keyNum[5000] + n1000(num - 5);
	    } else if (num == 9){
	        return n1000(1) + keyNum[10000];
	    }
	}

	function convert(num) {
	    var tenhundred = 0;
	    var hundred = 0;
	    var ten = 0;
	    var a0 = 0;
	    tenhundred = Math.floor(num / 1000);
	    num -= tenhundred * 1000;
	    hundred = Math.floor(num / 100);
	    num -= hundred * 100;
	    ten = Math.floor(num / 10);
	    a0 = num - ten * 10;
	    return tenhundreds(tenhundred) + hundreds(hundred) + ten10(ten) + units(a0);
	}


	convert(36);
